// --- HASH FUNCTION ---
async function sha256(message) {
  const msgBuffer = new TextEncoder().encode(message);                   
  const hashBuffer = await crypto.subtle.digest('SHA-256', msgBuffer);   
  const hashArray = Array.from(new Uint8Array(hashBuffer));              
  return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
}

// --- LOGIN ---
document.getElementById('roleSelect').addEventListener('change', function(){
  if(this.value==='admin'){
    document.getElementById('adminPassInput').classList.remove('hidden');
  } else {
    document.getElementById('adminPassInput').classList.add('hidden');
  }
});

async function login(){
  role = document.getElementById('roleSelect').value;
  let name = document.getElementById('userNameInput').value.trim();
  let pass = document.getElementById('adminPassInput').value;
  if(!role || !name){document.getElementById('loginMsg').innerText='Fill all fields'; return;}
  if(role==='admin'){
    const hash = await sha256(pass);
    if(hash!==adminHash){document.getElementById('loginMsg').innerText='Wrong admin password'; return;}
    currentUser = name;
    currentId = 'admin-'+name;
    document.getElementById('dashboardTabBtn').classList.remove('hidden');
    document.getElementById('settingsTabBtn').classList.remove('hidden');
    document.getElementById('adminControls').classList.remove('hidden');
    document.getElementById('paymentContainer').classList.add('hidden');
  } else {
    currentUser = name;
    if(!students[name]){students[name]={id:crypto.randomUUID(),lessons:[],paid:false};}
    currentId = students[name].id;
    document.getElementById('dashboardTabBtn').classList.remove('hidden');
    document.getElementById('settingsTabBtn').classList.add('hidden');
    document.getElementById('adminControls').classList.add('hidden');
    document.getElementById('paymentContainer').classList.remove('hidden');
  }
  localStorage.setItem('students', JSON.stringify(students));
  document.getElementById('welcomeMsg').innerText = `Welcome, ${currentUser}!`;
  renderLessons();
  openTab('dashboard');
}

// --- TABS ---
function openTab(tabName){
  document.querySelectorAll('.tab-content').forEach(t=>t.style.display='none');
  document.querySelectorAll('.tab-btn').forEach(b=>b.classList.remove('active'));
  document.getElementById(tabName).style.display='block';
  let tabs = document.querySelectorAll('.tab-btn');
  tabs.forEach(t=>{ if(t.innerText.toLowerCase().includes(tabName)) t.classList.add('active'); });
}

// --- LESSONS ---
function addLesson(){
  const name = document.getElementById('lessonNameInput').value.trim();
  const link = document.getElementById('lessonLinkInput').value.trim();
  const cat = document.getElementById('lessonCategoryInput').value;
  if(!name||!link){alert('Fill lesson name and link'); return;}
  if(!link.match(/\.(pdf|mp4|webm|ogg)$/)){alert('Only PDF or video links allowed'); return;}
  lessons.push({name, link, category:cat, status:'Not Done'});
  localStorage.setItem('lessons', JSON.stringify(lessons));
  renderLessons();
  document.getElementById('lessonNameInput').value='';
  document.getElementById('lessonLinkInput').value='';
}
function deleteLesson(index){
  if(confirm('Delete this lesson?')){
    lessons.splice(index,1);
    localStorage.setItem('lessons', JSON.stringify(lessons));
    renderLessons();
  }
}
function toggleLessonStatus(index){
  let lesson = lessons[index];
  lesson.status = lesson.status==='Done'?'Not Done':'Done';
  localStorage.setItem('lessons', JSON.stringify(lessons));
  renderLessons();
}
function renderLessons(){
  if(!currentUser) return;
  const container = document.getElementById('lessonsContainer');
  container.innerHTML = '';
  lessons.forEach((l,i)=>{
    const div = document.createElement('div');
    let statusClass = l.status==='Done'?'done':(role==='student' && !students[currentUser].paid?'unpaid':'notdone');
    div.className = 'lesson-card '+statusClass;
    let buttons = '';
    if(role==='admin') buttons=`<button onclick="toggleLessonStatus(${i})">${l.status}</button><button onclick="deleteLesson(${i})">Delete</button>`;
    else buttons=`<button onclick="toggleLessonStatus(${i})">${l.status}</button>`;
    div.innerHTML=`<div class="card-content"><strong>${l.name}</strong> [${l.category}]<br><a href="${l.link}" target="_blank">Open Lesson</a>
      <div class="progress-bar"><div class="progress-fill" style="width:${l.status==='Done'?100:0}%"></div></div></div>
    <div class="card-buttons">${buttons}</div>`;
    container.appendChild(div);
  });
}

function payNow(){
  if(!settings.stripe){alert('Payment link not set by admin'); return;}
  // Simulate payment success for tablet-only testing
  students[currentUser].paid = true;
  localStorage.setItem('students', JSON.stringify(students));
  alert('Payment Successful!');
  renderLessons();
}

// --- BACKUP ---
function exportData(){
  const dataStr = JSON.stringify({students,lessons,settings,aboutInfo});
  const blob = new Blob([dataStr],{type:"application/json"});
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'elf_backup.json';
  a.click();
}
function importData(){
  const fileInput = document.getElementById('importFile');
  if(fileInput.files.length===0){alert('Select a file'); return;}
  const file = fileInput.files[0];
  const reader = new FileReader();
  reader.onload = function(e){
    const data = JSON.parse(e.target.result);
    students = data.students || {};
    lessons = data.lessons || [];
    settings = data.settings || settings;
    aboutInfo = data.aboutInfo || aboutInfo;
    localStorage.setItem('students',JSON.stringify(students));
    localStorage.setItem('lessons',JSON.stringify(lessons));
    localStorage.setItem('settings',JSON.stringify(settings));
    localStorage.setItem('aboutInfo',JSON.stringify(aboutInfo));
    renderLessons();
    renderAbout();
    alert('Data imported successfully!');
  };
  reader.readAsText(file);
}
